Curso de JavaScript con Ejemplos

1. Variables y constantes

Ejemplo de var, let, y const:

var nombre = "Juan"; // Variable global o de bloque
let edad = 25;       // Variable local
const pais = "México"; // Constante inmutable

console.log(nombre, edad, pais);

2. Tipos de datos primitivos

Ejemplo de tipos de datos:

let texto = "Hola";      // String
let numero = 123;         // Number
let esVerdad = true;      // Boolean
let indefinido;           // Undefined
let nulo = null;          // Null

console.log(typeof texto); // "string"
console.log(typeof numero); // "number"

3. Operadores

Ejemplo de operadores:

let suma = 10 + 5; // Suma
let resta = 10 - 5;
let esIgual = suma === resta; // Comparación estricta

console.log(suma, resta, esIgual);

4. Condicionales

Ejemplo de condicionales:

let edad = 18;

if (edad >= 18) {
    console.log("Mayor de edad");
} else {
    console.log("Menor de edad");
}

5. Bucles

Ejemplo de bucles:

for (let i = 0; i < 5; i++) {
    console.log("Iteración:", i);
}

let j = 0;
while (j < 3) {
    console.log("While:", j);
    j++;
}

6. Arrays

Ejemplo de arrays:

let frutas = ["Manzana", "Banana"];
frutas.push("Naranja");
console.log(frutas);

// Ejemplo con map
let longitudes = frutas.map(fruta => fruta.length);
console.log(longitudes);

// Ejemplo con filter
let filtradas = frutas.filter(fruta => fruta.includes("a"));
console.log(filtradas);

// Ejemplo con reduce
let concatenadas = frutas.reduce((acumulado, fruta) => acumulado + fruta, "");
console.log(concatenadas);

7. Objetos

Ejemplo de objetos:

let persona = {
    nombre: "Carlos",
    edad: 30,
    saludar: function() {
        console.log("Hola, soy " + this.nombre);
    }
};

persona.saludar();

// Acceso y modificación de propiedades
persona.edad = 31;
console.log(persona);

8. Funciones

Ejemplo de funciones predefinidas y personalizadas:

function sumar(a, b) {
    return a + b;
}

let resultado = sumar(10, 15);
console.log(resultado);

// Funciones predefinidas
let texto = "JavaScript";
console.log(texto.toUpperCase());
console.log(Math.max(5, 10, 15));

9. Promesas

Ejemplo de promesas:

let promesa = new Promise((resolve, reject) => {
    setTimeout(() => resolve("Promesa cumplida"), 1000);
});

promesa.then(mensaje => console.log(mensaje));

// Ejemplo con async/await
async function obtenerDatos() {
    let datos = await promesa;
    console.log(datos);
}

obtenerDatos();

10. Manipulación del DOM

Ejemplo de manipulación del DOM:

let boton = document.createElement("button");
let texto = document.createTextNode("Presióname");
boton.appendChild(texto);
document.body.appendChild(boton);

boton.addEventListener("click", () => {
    alert("Botón presionado");
});

11. Fetch

Ejemplo de uso de fetch para obtener datos de una API:

fetch("https://jsonplaceholder.typicode.com/posts")
    .then(response => {
        if (!response.ok) {
            throw new Error("Error en la respuesta de la API");
        }
        return response.json();
    })
    .then(data => {
        console.log("Datos obtenidos:", data);
    })
    .catch(error => {
        console.error("Error al obtener los datos:", error);
    });
